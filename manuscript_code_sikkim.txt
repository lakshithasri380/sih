            document.querySelectorAll('.content-section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.nav-button').forEach(b => b.classList.remove('active'));
            
            document.getElementById(section).classList.add('active');
            event.target.classList.add('active');
            
            currentSection = section;
            
            if (section !== 'about') {
                populateGrid(section);
            }
        }

        // Populate archive grid
        function populateGrid(section) {
            const grid = document.getElementById(section + 'Grid');
            const data = archiveData[section] || [];
            
            grid.innerHTML = '';
            
            let filteredData = currentFilter === 'all' ? data : data.filter(item => item.type === currentFilter);
            
            // Apply search filter if search input is not empty
            const query = document.getElementById('searchInput').value.toLowerCase().trim();
            if (query) {
                filteredData = filteredData.filter(item => 
                    item.title.toLowerCase().includes(query) ||
                    item.description.toLowerCase().includes(query) ||
                    item.location.toLowerCase().includes(query)
                );
            }
            
            filteredData.forEach(item => {
                const itemElement = createArchiveItem(item);
                grid.appendChild(itemElement);
            });
            
            if (filteredData.length === 0) {
                grid.innerHTML = '<div class="loading">✨ No mystical treasures found matching the current filter ✨</div>';
            }
        }

        // Create archive item element
        function createArchiveItem(item) {
            const div = document.createElement('div');
            div.className = 'archive-item';
            div.onclick = () => openModal(item);
            
            // Determine image class based on item type and title
            let imageClass = '';
            if (item.title.includes('Pemayangtse')) {
                imageClass = 'monastery-pemayangtse';
            } else if (item.title.includes('Rumtek')) {
                imageClass = 'monastery-rumtek';
            } else if (item.title.includes('Tashiding')) {
                imageClass = 'monastery-tashiding';
            } else if (item.title.includes('Enchey')) {
                imageClass = 'monastery-enchey';
            } else if (item.title.includes('Dubdi')) {
                imageClass = 'monastery-dubdi';
            } else if (item.title.includes('Sangachoeling')) {
                imageClass = 'monastery-sangachoeling';
            } else if (item.title.includes('Ralong')) {
                imageClass = 'monastery-ralong';
            } else if (item.title.includes('Phensang')) {
                imageClass = 'monastery-phensang';
            } else if (item.title.includes('Royal') || item.title.includes('Palace') || item.title.includes('Chogyal')) {
                imageClass = 'royal-palace';
            } else if (item.type === 'Cultural' || item.title.includes('Lepcha') || item.title.includes('Village')) {
                imageClass = 'cultural-heritage';
            } else if (currentSection === 'manuscripts') {
                imageClass = 'manuscript-scroll';
            } else if (currentSection === 'documents') {
                imageClass = 'historical-document';
            } else {
                const monasteries = ['monastery-pemayangtse', 'monastery-rumtek', 'monastery-tashiding', 'monastery-enchey', 'monastery-dubdi', 'monastery-sangachoeling', 'monastery-ralong', 'monastery-phensang'];
                imageClass = monasteries[Math.floor(Math.random() * monasteries.length)];
            }
            
            div.innerHTML = `
                <div class="archive-image ${imageClass}">
                    <div style="text-align: center; z-index: 2; position: relative; background: rgba(0,0,0,0.6); padding: 0.5rem; border-radius: 10px; backdrop-filter: blur(5px);">
                        <div style="font-size: 0.9rem; opacity: 0.9; color: #fff;">🔮 Click to unveil mysteries 🔮</div>
                    </div>
                </div>
                <div class="archive-title">${item.title}</div>
                <div class="archive-description">${item.description}</div>
                <div class="archive-meta">
                    <span class="archive-date">${item.date}</span>
                    <span class="archive-type">${item.type}</span>
                </div>
            `;
            
            return div;
        }

        // Filter functionality
        function filterByType(type) {
            currentFilter = type;
            
            document.querySelectorAll('.filter-tag').forEach(tag => tag.classList.remove('active'));
            event.target.classList.add('active');
            
            populateGrid(currentSection);
        }

        // Search functionality
        function searchArchives() {
            populateGrid(currentSection);
        }

        // Enter key search
        document.getElementById('searchInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                searchArchives();
            }
        });

        // Animate stats on load
        document.addEventListener('DOMContentLoaded', function() {
            setTimeout(() => {
                const statNumbers = document.querySelectorAll('.stat-number');
                statNumbers.forEach((stat, index) => {
                    const finalNumber = parseInt(stat.textContent);
                    stat.textContent = '0';
                    
                    setTimeout(() => {
                        animateNumber(stat, 0, finalNumber, 2000);
                    }, index * 300);
                });
            }, 500);
        });

        function animateNumber(element, start, end, duration) {
            const range = end - start;
            const startTime = Date.now();
            
            function updateNumber() {
                const elapsedTime = Date.now() - startTime;
                const progress = Math.min(elapsedTime / duration, 1);
                const easeOutProgress = 1 - Math.pow(1 - progress, 3); // Ease out cubic
                const currentNumber = Math.floor(start + (range * easeOutProgress));
                
                element.textContent = currentNumber;
                
                if (progress < 1) {
                    requestAnimationFrame(updateNumber);
                }
            }
            
            updateNumber();
        }

        // Add mystical hover effects
        document.addEventListener('mousemove', function(e) {
            const particles = document.querySelectorAll('.particle');
            particles.forEach((particle, index) => {

                    particle.style.top = e.clientY + (Math.random() - 0.5) * 100 + 'px';
                }
            });
        });

            }
        });
